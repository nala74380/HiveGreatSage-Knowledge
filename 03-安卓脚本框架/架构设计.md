# 子系统3：安卓端脚本框架 — 架构设计

> **代号：** AndroidScript  
> **仓库：** HiveGreatSage-AndroidScript（框架基座）  
> **编辑器：** Cursor（代码编写）+ 懒人精灵 Win IDE（调试/打包）  
> **运行环境：** 懒人精灵高级版（**仅 root / 激活模式**，无障碍模式不支持）  
> **脚本语言：** Lua 5.3  
> **产出：** 每个游戏基于框架 fork 后，打包为独立 `.lrj`（热更新包）或 `.apk`

---

## 一、已确认的核心决策

```
✓ 运行模式：root / 激活 模式（无障碍分支全部删除，不保留）
✓ OCR 引擎：TomatoOCR 收费版（APK 方式加载，LuaEngine.loadApk）
✓ 开发基准分辨率：1280×720，支持 1920×1080 / 2560×1440 自动缩放
  · 三档均为 16:9，线性缩放无畸变，截图素材一套通用
  · 统一 DPI 320
✓ UI 方案：动态UI（ui.* API）或 H5（WebView），游戏定制层按需选择
✓ 局域网通信：WebSocket（startWebSocket，仅支持 ws://，不支持 wss://）
✓ 热更新格式：.lrj（installLrPkg，注意 Lua 区分大小写）
✓ 仓库策略：独立仓库模式（每个游戏 fork 框架基座）
✓ 代理/VPN：框架内置多客户端支持，配置驱动，用户按需切换
```

## 二、懒人精灵 IDE 固定结构 vs 框架逻辑结构

懒人精灵高级版 IDE 强制四个顶级目录，框架必须在此约束内展开。  
`require()` 路径基准为 `getWorkPath()`（即 `脚本/` 目录），支持子目录引用：

```lua
require("framework/verify")      -- 加载 脚本/framework/verify.lua
require("game/tasks/daily")      -- 加载 脚本/game/tasks/daily.lua
```

**映射关系：**

|IDE 固定目录|用途|框架映射|
|---|---|---|
|`脚本/`|所有 `.lua` 文件|框架核心层 + 游戏定制层（含子目录）|
|`界面/`|`.ui` 动态界面文件|动态UI面板（参数设置等）|
|`资源/`|`.rc` 资源包（ZIP容器）|APK引擎 + 截图模板 + H5文件|
|`插件/`|`.luaej` 加密插件|预留（暂无）|

### 4.2 `framework/proxy.lua` — 代理/VPN 多客户端管理

**设计原则：** 配置驱动，用户在 `config.lua` 填写 `PROXY_CLIENT` 字段即可切换，脚本其余部分无需改动。

```lua
-- 脚本/framework/proxy.lua
-- 代理/VPN 多客户端管理
-- 支持多种加速器/代理软件，配置驱动，互不干扰

local Config = require("config")
local Proxy  = {}

-- =====================================================================
-- 已支持客户端注册表
-- 新增客户端：在此表追加一条记录即可，其余代码无需修改
-- =====================================================================
local PROXY_CLIENTS = {

    -- 99手游加速器
    ["99"] = {
        name        = "99手游加速器",
        package     = "com.excean.na",
        -- 启动后需要点击的"开始加速"按钮颜色特征（1280×720基准）
        -- ⚠️ 实际坐标/颜色需要对应游戏调试后填入 ui_map.lua
        start_fn    = nil,   -- 由游戏定制层 ui_map 提供
    },

    -- BiuBiu加速器
    ["biubiu"] = {
        name        = "BiuBiu加速器",
        package     = "com.njh.biubiu",
        start_fn    = nil,
    },

    -- Kitsunebi 代理软件
    ["kitsunebi"] = {
        name        = "Kitsunebi",
        package     = "fun.kitsunebi.kitsunebi4android",
        start_fn    = nil,
    },

    -- OpenVPN
    ["openvpn"] = {
        name        = "OpenVPN",
        package     = "de.blinkt.openvpn",
        start_fn    = nil,
    },

    -- ----------------------------------------------------------------
    -- 预留扩展位（后续按需追加，格式同上）
    -- ----------------------------------------------------------------
    -- ["uu"]    = { name="UU加速器",   package="com.netease.uu",   start_fn=nil },
    -- ["xunyou"]= { name="迅游加速器", package="com.xunyou.mobile",start_fn=nil },
    -- ["nn"]    = { name="NN加速器",   package="com.nn.gameacc",   start_fn=nil },
    -- ["v2ray"] = { name="v2rayNG",    package="com.v2ray.ang",    start_fn=nil },
    -- ["clash"] = { name="ClashMeta",  package="com.github.metacubex.clash.meta", start_fn=nil },
}

-- =====================================================================
-- 公共接口
-- =====================================================================

--- 检查当前配置的代理客户端是否已安装
function Proxy.is_installed()
    local key = Config.PROXY_CLIENT
    if not key or key == "" then return true end  -- 未配置 = 不需要
    local client = PROXY_CLIENTS[key]
    if not client then
        print("[Proxy] 未知的代理客户端：" .. key)
        return false
    end
    local apps = getInstalledApk()
    if apps and string.find(tostring(apps), client.package, 1, true) then
        return true
    end
    return false
end

--- 启动代理并等待连接
-- @return boolean 是否成功（未配置时直接返回 true）
function Proxy.start()
    local key = Config.PROXY_CLIENT
    if not key or key == "" then return true end

    local client = PROXY_CLIENTS[key]
    if not client then
        print("[Proxy] 未知客户端：" .. key .. "，跳过代理启动")
        return false
    end

    print("[Proxy] 启动代理：" .. client.name .. "（" .. client.package .. "）")

    -- 检查是否已在运行
    if appIsRunning(client.package) and appIsFront(client.package) == false then
        -- 已在后台运行，大概率已连接，直接继续
        print("[Proxy] 代理已在后台运行，跳过启动")
        return true
    end

    -- 启动应用
    runApp(client.package)
    sleep(2000)

    -- 如果游戏定制层提供了"点击连接"的操作函数，执行它
    if client.start_fn and type(client.start_fn) == "function" then
        client.start_fn()
    end

    -- 等待代理连接
    local wait = (Config.PROXY_WAIT or 8) * 1000
    print("[Proxy] 等待代理连接 " .. (wait / 1000) .. " 秒...")
    sleep(wait)

    -- 按 Home 键切回桌面（为后续启动游戏做准备）
    keyPress(3)
    sleep(1000)

    print("[Proxy] 代理启动完成")
    return true
end

--- 停止代理客户端
function Proxy.stop()
    local key = Config.PROXY_CLIENT
    if not key or key == "" then return end
    local client = PROXY_CLIENTS[key]
    if not client then return end
    if appIsRunning(client.package) then
        stopApp(client.package)
        print("[Proxy] 代理已停止：" .. client.name)
    end
end

--- 注册游戏定制的"连接"操作（由 game/ui_map.lua 调用）
-- @param key     string  客户端标识（如 "biubiu"）
-- @param start_fn function 点击连接的操作函数
function Proxy.register_start_fn(key, start_fn)
    if PROXY_CLIENTS[key] then
        PROXY_CLIENTS[key].start_fn = start_fn
    end
end

--- 获取当前已知的所有客户端列表（调试用）
function Proxy.list()
    local list = {}
    for k, v in pairs(PROXY_CLIENTS) do
        table.insert(list, string.format("  [%s] %s → %s", k, v.name, v.package))
    end
    table.sort(list)
    return table.concat(list, "\n")
end

return Proxy
```

### 4.3 `framework/ocr.lua` — TomatoOCR 适配器

**TomatoOCR 加载方式（已确认）：**

```lua
-- APK 放入 资源/.rc 文件内
-- LuaEngine.loadApk("TomatoOCR.apk") 直接从 .rc 内加载，无需手动释放
local loader = LuaEngine.loadApk("TomatoOCR.apk")
local OCRApi = loader.loadClass("com.tomato.ocr.lr.OCRApi")
local tmo_ocr = OCRApi.init(LuaEngine.getContext())
tmo_ocr.setLicense("LICENSE|SECRET")
```

**适配器接口：**

```lua
-- 脚本/framework/ocr.lua

local Config = require("config")
local OCR    = {}
local _engine = nil

-- 初始化（main.lua 启动时调用一次）
-- @return boolean
function OCR.init()
    import('com.nx.assist.lua.LuaEngine')
    local loader = LuaEngine.loadApk("TomatoOCR.apk")
    if not loader then
        print("[OCR] loadApk 失败")
        return false
    end
    local OCRApi = loader.loadClass("com.tomato.ocr.lr.OCRApi")
    _engine = OCRApi.init(LuaEngine.getContext())

    if not _engine.setLicense(Config.OCR_LICENSE) then
        print("[OCR] License 验证失败")
        return false
    end
    _engine.setRecType(Config.OCR_REC_TYPE)
    _engine.setDetBoxType("rect")
    _engine.setDetUnclipRatio(1.9)
    _engine.setRecScoreThreshold(0.3)
    _engine.setReturnType("json")
    _engine.setBinaryThresh(0)
    print("[OCR] 初始化成功，模型：" .. Config.OCR_REC_TYPE)
    return true
end

-- 核心接口：区域识别
-- @param l,t,r,b 基于基准分辨率的坐标；全0=全屏
-- @return table[] { label, x, y, score }
function OCR.detect(l, t, r, b)
    if not _engine then return {} end
    local ok, result = pcall(function()
        local s = _engine.detect(l or 0, t or 0, r or 0, b or 0)
        return jsonLib.decode(s)
    end)
    return (ok and result) or {}
end

-- 便捷：查找文字，返回中心坐标
-- @return x, y 或 -1, -1
function OCR.find_text(text, l, t, r, b)
    for _, item in ipairs(OCR.detect(l, t, r, b)) do
        if item.label and string.find(item.label, text, 1, true) then
            return item.x, item.y
        end
    end
    return -1, -1
end

-- 便捷：获取区域内全部文字（拼接字符串）
function OCR.get_text(l, t, r, b)
    local parts = {}
    for _, item in ipairs(OCR.detect(l, t, r, b)) do
        if item.label then table.insert(parts, item.label) end
    end
    return table.concat(parts, " ")
end

return OCR
```

### 4.4 `framework/screen.lua` — 分辨率适配

```lua
-- 脚本/framework/screen.lua

local Config = require("config")
local Screen = {}

-- 启动时调用一次
function Screen.init()
    local w, h = getDisplaySize()
    print(string.format("[Screen] 实际 %dx%d，基准 %dx%d",
        w, h, Config.BASE_WIDTH, Config.BASE_HEIGHT))
    if w ~= Config.BASE_WIDTH or h ~= Config.BASE_HEIGHT then
        -- scale=0：输入坐标缩放，返回坐标保持真实值
        setScreenScale(1, Config.BASE_WIDTH, Config.BASE_HEIGHT, 0)
        print("[Screen] 分辨率缩放已启用")
    end
end

function Screen.get_size()
    return getDisplaySize()
end

return Screen
```

### 4.5 `main.lua` — 标准启动序列

```lua
-- 脚本/main.lua

local Config       = require("config")
local Screen       = require("framework/screen")
local Proxy        = require("framework/proxy")
local OCR          = require("framework/ocr")
local Verify       = require("framework/verify")
local Updater      = require("framework/updater")
local CloudSync    = require("framework/cloud_sync")
local Scheduler    = require("framework/scheduler")
local Logger       = require("framework/logger")
local ErrorHandler = require("framework/error_handler")

-- ======= 停止回调 =======
setStopCallBack(function(err, exitcode)
    Logger.info("停止 err=" .. tostring(err) .. " code=" .. tostring(exitcode))
    releaseCapture()
    -- ⚠️ 停止回调里不能 stopApp 代理，否则可能影响其他业务
end)

-- ======= 启动自检（环境探针） =======
print("=== 启动自检 ===")
print("workPath="   .. getWorkPath())
print("sdPath="     .. getSdPath())
print("apkVer="     .. getApkVerInt())
if getCpuArch then
    local t = getCpuArch()
    print("cpuArch=" .. (t==2 and "arm64" or t==1 and "arm" or "x86"))
end
-- 确认运行在 root/激活 模式
local env_type = getRunEnvType()
if env_type ~= 0 then
    toast("⚠️ 请使用 root 或激活模式运行", 0, 0, 20)
    sleep(3000)
    exitScript()
end
print("runEnv=root/激活 ✓")

-- ======= 初始化序列 =======
Screen.init()
Logger.init()

-- 代理/VPN（按 config 配置决定是否启动）
Proxy.start()

-- OCR 初始化
if not OCR.init() then
    toast("OCR 初始化失败，检查 License", 0, 0, 20)
    sleep(3000)
    exitScript()
end

-- 授权验证
if not Verify.login() then
    toast("授权验证失败", 0, 0, 20)
    sleep(3000)
    exitScript()
end

-- 热更新检查（如有新版本，自动 restartScript()）
Updater.check_and_update()

-- 拉取云端参数
CloudSync.pull_params()

-- ======= 启动任务调度器 =======
Scheduler.run()
```
